);

  "dept_no" TEXT,
  "from_date" TEXT,
  "to_date" TEXT

"emp_no" INTEGER,
  "dept_no" TEXT,
  "from_date" TEXT,
  "to_date" TEXT
);
CREATE TABLE IF NOT EXISTS "salaries" (

  "to_date" TEXT
);

  "title" TEXT,
